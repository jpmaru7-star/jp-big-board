  const tbody = document.querySelector("#board tbody");
  const searchEl = document.getElementById("search");
  const posFilter = document.getElementById("posFilter");
  const countEl = document.getElementById("count");
  const resetBtn = document.getElementById("resetBtn");
  const headers = document.querySelectorAll("thead th");

  let sortKey = "rank";
  let sortDir = "asc";

  function uniq(arr){ return [...new Set(arr)].sort(); }

  function normalize(s){ return (s ?? "").toString().toLowerCase().trim(); }

  function buildPosOptions(){
    const positions = uniq(players.map(p => p.pos));
    positions.forEach(pos => {
      const opt = document.createElement("option");
      opt.value = pos;
      opt.textContent = pos;
      posFilter.appendChild(opt);
    });
  }

  function compare(a,b,key){
    const va = a[key], vb = b[key];
    const na = Number(va), nb = Number(vb);

    // numeric sort if possible
    if (!Number.isNaN(na) && !Number.isNaN(nb) && (va !== "" && vb !== "")) {
      return na - nb;
    }

    // fallback string sort
    return (va ?? "").toString().localeCompare((vb ?? "").toString(), undefined, {numeric:true, sensitivity:"base"});
  }

  function getFiltered(){
    const q = normalize(searchEl.value);
    const pos = posFilter.value;

    return players.filter(p => {
      const matchesPos = (pos === "ALL") ? true : p.pos === pos;
      const hay = normalize(`${p.rank} ${p.name} ${p.pos} ${p.school} ${p.grade} ${p.notes}`);
      const matchesSearch = q ? hay.includes(q) : true;
      return matchesPos && matchesSearch;
    });
  }

  function render(){
    const rows = getFiltered()
      .slice()
      .sort((a,b) => {
        const c = compare(a,b,sortKey);
        return sortDir === "asc" ? c : -c;
      });

    tbody.innerHTML = "";
    rows.forEach(p => {
      const tr = document.createElement("tr");
      tr.innerHTML = `
        <td class="rank">${p.rank}</td>
        <td class="name">
          ${p.name}
          <span class="badge">#${p.rank}</span>
        </td>
        <td class="pos">${p.pos}</td>
        <td class="school">${p.school}</td>
        <td class="grade">${p.grade || "—"}</td>
        <td class="notes">${p.notes || "—"}</td>
      `;
      tbody.appendChild(tr);
    });

    countEl.textContent = rows.length;
  }

  headers.forEach(th => {
    th.addEventListener("click", () => {
      const key = th.dataset.key;
      if (!key) return;
      if (sortKey === key) {
        sortDir = (sortDir === "asc") ? "desc" : "asc";
      } else {
        sortKey = key;
        sortDir = "asc";
      }
      render();
    });
  });

  searchEl.addEventListener("input", render);
  posFilter.addEventListener("change", render);
  resetBtn.addEventListener("click", () => {
    searchEl.value = "";
    posFilter.value = "ALL";
    sortKey = "rank";
    sortDir = "asc";
    render();
  });

  // Export function (optional)
  function downloadJSON(){
    const data = getFiltered();
    const blob = new Blob([JSON.stringify(data, null, 2)], {type:"application/json"});
    const url = URL.createObjectURL(blob);
    const a = document.createElement("a");
    a.href = url;
    a.download = "jp-big-board.json";
    document.body.appendChild(a);
    a.click();
    a.remove();
    URL.revokeObjectURL(url);
  }
  window.downloadJSON = downloadJSON;
